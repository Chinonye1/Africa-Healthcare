Python Code Example

This is an example of a Python code for the image retrieval:

                        
                            import requests
                            import os
                    
                            # Function to fetch and download images from an API
                            def fetch_images(api_key, query, count=10):
                                base_url = 'https://api.unsplash.com/photos/random'
                                headers = {'Authorization': f'Client-ID {api_key}'}
                                params = {'query': query, 'count': count}
                    
                                response = requests.get(base_url, headers=headers, params=params)
                                images = response.json()
                    
                                return images
                    
                            # Function to save images to a local directory
                            def save_images(images, save_path):
                                os.makedirs(save_path, exist_ok=True)
                    
                                for i, image in enumerate(images):
                                    image_url = image['urls']['regular']
                                    response = requests.get(image_url)
                    
                                    with open(os.path.join(save_path, f'image_{i+1}.jpg'), 'wb') as f:
                                        f.write(response.content)
                    
                            # Example usage
                            api_key = 'your_unsplash_api_key'
                            query = 'beautiful places in Africa'
                            image_count = 5
                            save_directory = 'images'
                    
                            images = fetch_images(api_key, query, image_count)
                            save_images(images, save_directory)
